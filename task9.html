Task 9: Paging Technique

#include<stdio.h>
void main() {
    int np, ps, pt[20], nf, la, pn, index, pa, i;
    
    printf("\nEnter number of pages:");
    scanf("%d",&np);
    printf("\nEnter page size:");
    scanf("%d",&ps);
    printf("\nEnter page table:");
    
    for(i=0;i<np;i++)
        scanf("%d",&pt[i]);
    
    printf("\nEnter number of frames:");
    scanf("%d",&nf);
    printf("\nEnter logical address:");
    scanf("%d",&la);
    
    pn = la/ps;
    index = la%ps;
    pa = (pt[pn]*ps) + index;
    
    printf("\nPhysical address is %d",pa);
}

Output:

Enter number of pages:4
Enter page size:300
Enter page table:4 2 1 3
Enter number of frames:4
Enter logical address:1000

Physical address is 401
