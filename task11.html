Task 11: Disk Scheduling Algorithms

#include<stdio.h>
#include<stdlib.h>

int absolute(int a, int b) {
    int c = a-b;
    return (c<0) ? -c : c;
}

int main() {
    int choice, m, n, x, start, i, j, pos, min, a[15], count;
    count = 0;
    
    printf("\nEnter the number of cylinders :");
    scanf("%d",&m);
    printf("\nEnter the number of requests :");
    scanf("%d",&n);
    printf("\nEnter current position :");
    scanf("%d",&start);
    
    printf("\nEnter the request queue :");
    for(i=0;i<n;i++) {
        scanf("%d",&a[i]);
        if(a[i]>=m) {
            printf("\ninvalid input");
            scanf("%d",&a[i]);
        }
    }
    
    do {
        printf("\nDISK SCHEDULING ALGORITHMS\n1. FCFS\n2. SSTF\n3. SCAN\n4. C-SCAN\n5. LOOK\n6. C-LOOK");
        printf("\nEnter choice :");
        scanf("%d",&choice);
        count=0;
        x=start;
        
        switch(choice) {
            case 1: // FCFS
                printf("\nFCFS :\n");
                printf("Scheduling services the request in order:\n%d\t",start);
                for(i=0;i<n;i++) {
                    count += absolute(x,a[i]);
                    x=a[i];
                    printf("%d\t",x);
                }
                printf("\nTotal Head Movement: %d Cylinders",count);
                break;
                
            case 2: // SSTF
                printf("\nSSTF:\n");
                printf("Scheduling services the request in order:\n%d\t",start);
                for(i=0;i<n;i++) {
                    min = absolute(a[i],x);
                    pos = i;
                    for(j=i;j<n;j++) {
                        if(min > absolute(x,a[j])) {
                            pos = j;
                            min = absolute(x,a[j]);
                        }
                    }
                    count += absolute(x,a[pos]);
                    x = a[pos];
                    a[pos] = a[i];
                    a[i] = x;
                    printf("%d\t",x);
                }
                printf("\nTotal Head Movement: %d Cylinders",count);
                break;
                
            case 3: // SCAN
                printf("\nSCAN:\n");
                printf("Scheduling services the request in order:\n");
                count = 0;
                pos = 0;
                
                // Sort requests
                for(i=0;i<n;i++)
                    for(j=0;j<n-i-1;j++)
                        if(a[j]>a[j+1]) {
                            x=a[j];
                            a[j]=a[j+1];
                            a[j+1]=x;
                        }
                
                // Find position
                for(i=0;i<n;i++)
                    if(a[i]<start) pos++;
                
                // Service requests before head
                for(i=pos-1;i>=0;i--) {
                    count += absolute(x,a[i]);
                    x = a[i];
                    printf("%d\t",x);
                }
                
                // Service requests after head
                for(i=pos;i<n;i++) {
                    count += absolute(x,a[i]);
                    x = a[i];
                    printf("%d\t",x);
                }
                printf("\nTotal Head Movement: %d Cylinders",count);
                break;
                
            // Additional cases for other algorithms would follow...
        }
        
        printf("\nContinue (1/0): ");
        scanf("%d",&choice);
    } while(choice==1);
    
    return 0;
}
